# Phase 1 Design: Data Model Extension

**Feature**: 003-inci-label-generator
**Design Date**: 2025-11-05
**Input**: Research findings from research.md

## Overview

Extend the Oil model with `saponified_inci_name` field to store post-saponification INCI nomenclature. This field enables efficient INCI label generation without runtime pattern matching for the 37 reference oils.

## Data Model Changes

### Oil Model Extension

**File**: `app/models/oil.py`

**New Field**:
```python
saponified_inci_name: Mapped[str | None] = mapped_column(
    String(200),
    nullable=True,
    comment="Post-saponification INCI name (e.g., 'Sodium Cocoate' or 'Potassium Cocoate')"
)
```

**Field Specifications**:
- **Type**: String(200) - matches existing `inci_name` field length for consistency
- **Nullable**: True - allows gradual backfill and handles oils without established saponified names
- **Comment**: Descriptive for database documentation
- **Storage**: Direct string value, not JSONB (simple lookup, no complex structure needed)

**Updated Model**:
```python
"""Oil model with SAP values and fatty acid profiles"""
from datetime import datetime

import sqlalchemy as sa
from sqlalchemy import String, Float, DateTime
from sqlalchemy.dialects.postgresql import JSONB
from sqlalchemy.orm import Mapped, mapped_column

from app.db.base import Base


class Oil(Base):
    """
    Oil/fat ingredient with saponification values and properties.

    SAP values define grams of lye needed per gram of oil.
    Fatty acid profile determines soap quality characteristics.
    Saponified INCI name represents post-saponification nomenclature.
    """

    __tablename__ = "oils"

    id: Mapped[str] = mapped_column(
        String(50),
        primary_key=True,
    )
    common_name: Mapped[str] = mapped_column(
        String(100),
        nullable=False,
    )
    inci_name: Mapped[str] = mapped_column(
        String(200),
        nullable=False,
    )
    saponified_inci_name: Mapped[str | None] = mapped_column(
        String(200),
        nullable=True,
        comment="Post-saponification INCI name (e.g., 'Sodium Cocoate' or 'Potassium Cocoate')"
    )
    sap_value_naoh: Mapped[float] = mapped_column(
        Float,
        nullable=False,
        comment="Grams NaOH per gram of oil",
    )
    sap_value_koh: Mapped[float] = mapped_column(
        Float,
        nullable=False,
        comment="Grams KOH per gram of oil",
    )
    iodine_value: Mapped[float] = mapped_column(
        Float,
        nullable=False,
        comment="Measure of unsaturation",
    )
    ins_value: Mapped[float] = mapped_column(
        Float,
        nullable=False,
        comment="Iodine Number Saponification (hardness indicator)",
    )
    fatty_acids: Mapped[dict] = mapped_column(
        JSONB,
        nullable=False,
        comment="Percentages of 8 fatty acids: lauric, myristic, palmitic, stearic, ricinoleic, oleic, linoleic, linolenic",
    )
    quality_contributions: Mapped[dict] = mapped_column(
        JSONB,
        nullable=False,
        comment="Contribution to 7 quality metrics: hardness, cleansing, conditioning, bubbly_lather, creamy_lather, longevity, stability",
    )
    created_at: Mapped[datetime] = mapped_column(
        DateTime(timezone=True),
        nullable=False,
        server_default=sa.func.now(),
        comment="UTC timestamp generated by PostgreSQL server",
    )
    updated_at: Mapped[datetime] = mapped_column(
        DateTime(timezone=True),
        nullable=False,
        server_default=sa.func.now(),
        comment="UTC timestamp auto-updated by PostgreSQL trigger",
    )

    def __repr__(self) -> str:
        return f"<Oil(id={self.id}, name={self.common_name})>"
```

## Data Population Strategy

### Reference Data Backfill

**Source**: `app/data/saponified-inci-terms.json`

**Backfill Script**: `scripts/backfill_saponified_inci_names.py`

```python
"""
Backfill saponified_inci_name from reference data

Usage: python scripts/backfill_saponified_inci_names.py
"""
import json
import asyncio
from pathlib import Path

from sqlalchemy import select, update
from sqlalchemy.ext.asyncio import AsyncSession

from app.db.base import async_session_maker
from app.models.oil import Oil


async def load_reference_data() -> dict:
    """Load saponified INCI reference data from JSON file"""
    reference_file = Path(__file__).parent.parent / "app" / "data" / "saponified-inci-terms.json"

    with open(reference_file, 'r') as f:
        data = json.load(f)

    # Build lookup dictionary: oil_id -> saponified_inci_naoh
    lookup = {}
    for oil_entry in data['oils']:
        lookup[oil_entry['oil_id']] = oil_entry['saponified_inci_naoh']

    return lookup


async def backfill_oils(session: AsyncSession, reference_lookup: dict) -> dict:
    """
    Backfill saponified_inci_name for oils in reference data

    Returns:
        dict: Statistics (updated_count, missing_count, total_count)
    """
    # Get all oils
    result = await session.execute(select(Oil))
    oils = result.scalars().all()

    updated_count = 0
    missing_count = 0

    for oil in oils:
        if oil.id in reference_lookup:
            # Update with reference data
            oil.saponified_inci_name = reference_lookup[oil.id]
            updated_count += 1
        else:
            # Log missing oils for manual review
            print(f"WARNING: No reference data for oil: {oil.id} ({oil.common_name})")
            missing_count += 1

    await session.commit()

    return {
        'updated_count': updated_count,
        'missing_count': missing_count,
        'total_count': len(oils)
    }


async def main():
    """Execute backfill operation"""
    print("Loading reference data...")
    reference_lookup = await load_reference_data()
    print(f"Loaded {len(reference_lookup)} oil entries from reference data")

    print("\nBackfilling saponified_inci_name values...")
    async with async_session_maker() as session:
        stats = await backfill_oils(session, reference_lookup)

    print("\n=== Backfill Complete ===")
    print(f"Total oils: {stats['total_count']}")
    print(f"Updated: {stats['updated_count']}")
    print(f"Missing reference data: {stats['missing_count']}")

    if stats['missing_count'] > 0:
        print("\nAction Required: Review oils without reference data and add manually or via pattern generation")


if __name__ == "__main__":
    asyncio.run(main())
```

**Execution Plan**:
1. Run migration: `alembic upgrade head`
2. Ensure reference data exists: `app/data/saponified-inci-terms.json`
3. Execute backfill: `python scripts/backfill_saponified_inci_names.py`
4. Review output for missing oils
5. Manually populate missing oils or use pattern generation fallback

## Pattern Generation Fallback

### For Oils Without Reference Data

**Service**: `app/services/inci_naming.py`

**Fallback Logic**:
```python
def generate_saponified_name(
    common_name: str,
    lye_type: Literal["naoh", "koh"]
) -> str:
    """
    Generate saponified INCI name using pattern-based approach

    Used when oil lacks reference data in saponified-inci-terms.json

    Args:
        common_name: Oil common name (e.g., "Mango Butter")
        lye_type: "naoh" for Sodium, "koh" for Potassium

    Returns:
        Generated saponified INCI name (e.g., "Sodium Mangoate")
    """
    # Remove common suffixes
    base_name = common_name.replace(" Oil", "").replace(" Butter", "").strip()

    # Choose prefix based on lye type
    prefix = "Sodium" if lye_type == "naoh" else "Potassium"

    # Generate saponified name
    # Note: This is a fallback - prefer reference data for accuracy
    saponified_name = f"{prefix} {base_name}ate"

    return saponified_name


def get_saponified_inci_name(
    oil: Oil,
    lye_type: Literal["naoh", "koh"]
) -> tuple[str, bool]:
    """
    Get saponified INCI name with fallback to pattern generation

    Args:
        oil: Oil model instance
        lye_type: "naoh" or "koh"

    Returns:
        tuple: (saponified_name, is_generated)
            - saponified_name: The INCI name (from DB or generated)
            - is_generated: True if pattern-generated, False if from reference data
    """
    # Check if DB has saponified name
    if oil.saponified_inci_name:
        # Adjust for lye type if needed (DB stores NaOH by default)
        if lye_type == "koh":
            # Replace "Sodium" with "Potassium"
            koh_name = oil.saponified_inci_name.replace("Sodium", "Potassium")
            return (koh_name, False)
        return (oil.saponified_inci_name, False)

    # Fallback to pattern generation
    generated_name = generate_saponified_name(oil.common_name, lye_type)
    return (generated_name, True)
```

## Data Integrity Constraints

### Validation Rules

**Application-Level Validation** (in Pydantic schemas):
```python
from pydantic import BaseModel, Field, validator

class OilCreate(BaseModel):
    """Schema for creating new oil with saponified INCI name"""
    id: str
    common_name: str
    inci_name: str
    saponified_inci_name: str | None = Field(
        None,
        description="Post-saponification INCI name",
        max_length=200
    )
    # ... other fields

    @validator('saponified_inci_name')
    def validate_saponified_name(cls, v):
        """Validate saponified INCI name format"""
        if v is None:
            return v

        # Check for proper prefix
        valid_prefixes = ['Sodium ', 'Potassium ']
        if not any(v.startswith(prefix) for prefix in valid_prefixes):
            raise ValueError(
                "Saponified INCI name must start with 'Sodium ' or 'Potassium '"
            )

        # Check for proper suffix
        if not v.endswith('ate'):
            raise ValueError("Saponified INCI name must end with 'ate'")

        return v
```

**Database Constraints**:
- No unique constraint (multiple oils could theoretically have same saponified name)
- Length constraint enforced by String(200)
- Nullable allowed for gradual adoption
